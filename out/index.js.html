/**
 * construct of indexItem
 * @constructor
 * @prop {String} title link title
 * @prop {String} path  link path
 * @see {Person}
 */
class IndexItem {
  constructor(title, path) {
    this.title = title;
    this.path = path;
  }
}

 return <li><Link to={item.path} >{item.title}</Link></li>
